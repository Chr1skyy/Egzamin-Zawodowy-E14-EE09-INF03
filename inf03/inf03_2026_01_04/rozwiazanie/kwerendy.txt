1. SELECT id, imie, nazwisko FROM matura_maturzysta WHERE szkola = 'T3' ORDER BY nazwisko ASC;
2. SELECT DISTINCT przedmiot FROM matura_arkusz;
3. SELECT MIN(rok), MAX(rok) FROM matura_arkusz;
4. SELECT maturzysta_id, AVG(punkty) AS 'Wynik' FROM matura_wynik GROUP BY maturzysta_id ORDER BY Wynik DESC LIMIT 1;
5. SELECT matura_arkusz.rok, matura_arkusz.sesja, matura_arkusz.przedmiot, matura_wynik.punkty FROM matura_arkusz JOIN matura_wynik USING symbol WHERE matura_wynik.maturzysta_id = 31;
